continue

: 关键字 退出本次(当前的循环) 继续执行剩余次数循环

for (var i = 1; i <= 5; i++) {

if (i == 3) {

continue; //只要遇见continue就退出本次循环 直接跳到i++

}

console.log('我正在吃第' + i + '个包子');

}

例子

1~100间,除了能被7整除之外的整数和

for (i = 1; i <= 100; i++) {

if (i % 7 == 0) {

continue;

}

sum += i;

}

console.log(sum);
